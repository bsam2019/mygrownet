## Technical Implementation Notes

### Database Schema Additions Needed:

```sql
-- Share certificates
CREATE TABLE investor_share_certificates (
    id BIGINT PRIMARY KEY,
    investor_account_id BIGINT,
    certificate_number VARCHAR(50) UNIQUE,
    shares_percentage DECIMAL(10,4),
    issue_date DATE,
    pdf_path VARCHAR(255),
    verification_hash VARCHAR(255),
    created_at TIMESTAMP
);

-- Dividends
CREATE TABLE investor_dividends (
    id BIGINT PRIMARY KEY,
    investor_account_id BIGINT,
    amount DECIMAL(15,2),
    payment_date DATE,
    tax_withheld DECIMAL(15,2),
    status ENUM('pending', 'paid', 'cancelled'),
    created_at TIMESTAMP
);

-- Shareholder votes
CREATE TABLE shareholder_resolutions (
    id BIGINT PRIMARY KEY,
    title VARCHAR(255),
    description TEXT,
    voting_start DATE,
    voting_end DATE,
    status ENUM('draft', 'active', 'closed'),
    created_at TIMESTAMP
);

CREATE TABLE shareholder_votes (
    id BIGINT PRIMARY KEY,
    resolution_id BIGINT,
    investor_account_id BIGINT,
    vote ENUM('for', 'against', 'abstain'),
    voted_at TIMESTAMP
);

-- Valuation history
CREATE TABLE company_valuations (
    id BIGINT PRIMARY KEY,
    valuation_amount DECIMAL(15,2),
    valuation_date DATE,
    valuation_method VARCHAR(100),
    notes TEXT,
    created_at TIMESTAMP
);
```
